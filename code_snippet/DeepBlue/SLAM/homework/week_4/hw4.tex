\section{矩阵微分}

\begin{enumerate}
    \item $\frac{d(Ax)}{dx} = A $
    \item $\frac{d(x^TAx)}{dx} = (A + A^T)x $
    \item $$xA^Tx = \sum_{i=1}^N \sum_{i=1}^N a_{ij}x_ix_j,$$
    
		$
			xx^T = 
			\left[
			\begin{array}{cccc}
			x_1^2 & x_1x_2 & ... & x_1x_n \\
			x_1x_2 & x_2^2 & ... & x_2x_n \\
			... & ... &   & ...\\
			x_1x_n & x_2x_n & ... & x_n^2
			\end{array}
			\right]
		$   
		
		对于$Axx^T$,对角线元素分别为
		$$\sum_{j=1}^N a_{1j}x_1x_j, \sum_{j=1}^N a_{2j}x_2x_j, ... , \sum_{j=1}^N a_{Nj}x_Nx_j$$
		所以，$$ tr(Axx^T) = \sum_{i=1}^N \sum_{j=1}^N a_{ij}x_ix_j$$
		由此可得，$xA^Tx = tr(Axx^T)$
    
    
\end{enumerate}
